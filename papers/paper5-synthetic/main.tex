\title{Synthetic Probability Type Theory\\[0.5em]
\large Towards Constructing Probability from Primitives}

\author{Probabilistic Foundations Project}

\maketitle

\begin{abstract}
We prove that probabilistic Zorn's lemma holds for finite types without any form of choice: every finite chain-complete probabilistic poset has an $\varepsilon$-near-maximal element. We then sketch how the infinite case follows from projective limits of finite approximations---constructing distributions rather than selecting points. Along the way, we reduce the axiomatic footprint of our probabilistic foundations from 48 to 43 axioms by observing that several axioms follow from commutativity, with further reductions possible given auxiliary lemmas. We conclude by posing open questions about synthetic probability type theory, where $\Prob$ would be a primitive type former.
\end{abstract}

\subsection{Derivable Axioms: Commutativity}

Five axioms are immediate consequences of commutativity:

\begin{proposition}[Trivial Reductions]
\label{prop:trivial-reductions}
The following five axioms are derivable from other axioms:\footnote{Each derivation is a one-line rewrite, formalized in \texttt{Synthetic.lean}.}
\begin{enumerate}
    \item $p \pmul \pone = p$ from $\pone \pmul p = p$ via $\pmul$-commutativity
    \item $p \padd \pzero = p$ from $\pzero \padd p = p$ via $\padd$-commutativity
    \item $p \pmul \pzero = \pzero$ from $\pzero \pmul p = \pzero$ via $\pmul$-commutativity
    \item $(p \padd q) \pmul r = (p \pmul r) \padd (q \pmul r)$ from $r \pmul (p \padd q) = (r \pmul p) \padd (r \pmul q)$ via $\pmul$-commutativity (right-distributivity from left-distributivity)
    \item $\pone \psub \pone = \pzero$ as the instance $p = \pone$ of $p \psub p = \pzero$
\end{enumerate}
\end{proposition}

\begin{proof}
Each follows by a single rewrite. For example:
\[
p \pmul \pone = \pone \pmul p = p
\]
where the first equality is $\pmul$-commutativity and the second is the $\mathsf{one\_mul}$ axiom.
\end{proof}

\textbf{Axiom count: $48 \to 43$.}

\subsection{De Morgan: Axiom vs.\ Theorem}

The De Morgan laws are currently axioms:
\begin{align}
\pone \psub (a \pmul b) &= (({\pone \psub a}) \padd ({\pone \psub b})) \psub (({\pone \psub a}) \pmul ({\pone \psub b})) \label{eq:dm-and} \\
\pone \psub ((a \padd b) \psub (a \pmul b)) &= ({\pone \psub a}) \pmul ({\pone \psub b}) \label{eq:dm-or}
\end{align}

\begin{proposition}[De Morgan in $[0,1]$]
If $\Prob$ is constructed as the real interval $[0,1]$, then \eqref{eq:dm-and} and \eqref{eq:dm-or} are theorems.
\end{proposition}

\begin{proof}
Standard algebra. For \eqref{eq:dm-and}: expand $(1-a) + (1-b) - (1-a)(1-b)$ to get $1 - ab$.
\end{proof}

\begin{remark}
This does \emph{not} derive De Morgan from our current axioms. It shows that \emph{if we construct} $\Prob$ as $[0,1]$, the axioms become theorems. Currently, De Morgan remains axiomatic.
\end{remark}

\textbf{Status: Axioms, pending $\Prob$ construction.}

\subsection{Conditional Expectation: Monotonicity from Positivity}

\begin{proposition}
Monotonicity of conditional expectation:
\[
(\forall x,\, f(x) \ple f'(x)) \Rightarrow \E[f | g] \ple \E[f' | g]
\]
is derivable from additivity plus a \emph{positivity} axiom:
\[
(\forall x,\, \pzero \ple f(x)) \Rightarrow \pzero \ple \E[f | g]
\]
\end{proposition}

\begin{proof}[Sketch]
Let $h = f' - f$ (pointwise). Then $h \geq 0$ and $f' = f + h$. By additivity:
\[
\E[f' | g] = \E[f | g] \padd \E[h | g]
\]
By positivity, $\E[h | g] \geq 0$, so $\E[f' | g] \geq \E[f | g]$.
\end{proof}

\begin{remark}
This trades the monotonicity axiom for a positivity axiom---not a net reduction in count. However, positivity is arguably more fundamental: it asserts that expectation preserves non-negativity, from which monotonicity follows. The value is conceptual clarity, not axiom count.
\end{remark}

\textbf{Status: Reorganization possible, not a net reduction.}

\section{Part 2: Finite Probabilistic Zorn}
\label{sec:finite-zorn}
% ============================================================================

We prove that probabilistic Zorn's lemma holds for finite types without any appeal to the Axiom of Choice. This is the foundation for the infinite case.

\subsection{Setup}

Recall from Paper 4:

\begin{definition}[Probabilistic Poset]
A \emph{probabilistic poset} on type $\alpha$ consists of:
\[
P.\mathsf{le} : \alpha \to \alpha \to \Prob
\]
with reflexivity ($P.\mathsf{le}(x,x) = \pone$) and transitivity.
\end{definition}

\begin{definition}[Strict Order]
$\plt(x,y) := P.\mathsf{le}(x,y) \pmul (\pone \psub P.\mathsf{le}(y,x))$
\end{definition}

\begin{definition}[$\varepsilon$-Near-Maximal]
Element $x$ is \emph{$\varepsilon$-near-maximal} if $\forall y,\, \plt(x,y) \ple \varepsilon$.
\end{definition}

\subsection{The Finite Case}

\begin{theorem}[Finite Probabilistic Zorn]
\label{thm:finite-zorn}
Let $\alpha$ be a finite nonempty type with probabilistic poset structure $P$. Then there exists $x \in \alpha$ and $\varepsilon \in \Prob$ such that $x$ is $\varepsilon$-near-maximal.
\end{theorem}

\begin{proof}
Since $\alpha$ is finite and nonempty, enumerate it as $\{a_1, \ldots, a_n\}$.

For each $a_i$, define the \emph{maximality defect}:
\[
\delta(a_i) := \max_{j \neq i} \plt(a_i, a_j)
\]
This measures how far $a_i$ is from being maximal: if $\delta(a_i) = \pzero$, then $a_i$ is truly maximal.

Since $\alpha$ is finite, the minimum exists:
\[
\varepsilon^* := \min_{i} \delta(a_i)
\]

Let $a_{i^*}$ be an element achieving this minimum. Then:
\[
\forall j,\, \plt(a_{i^*}, a_j) \leq \delta(a_{i^*}) = \varepsilon^*
\]

Thus $a_{i^*}$ is $\varepsilon^*$-near-maximal.
\end{proof}

\begin{remark}[Finite Suprema]
The proof assumes finite suprema (max/min) exist in $\Prob$. This follows from totality of $\ple$ and decidability of the ordering, or can be taken as an axiom for finite operations on $\Prob$.
\end{remark}

\begin{remark}[No Choice Used]
The proof uses only:
\begin{enumerate}
    \item Finiteness of $\alpha$ (to enumerate elements)
    \item Existence of max/min over finite sets (constructive for finite sets)
    \item Basic properties of $\Prob$
\end{enumerate}
No selection from infinite sets is required.
\end{remark}

\subsection{Strengthening for Classical Posets}

\begin{corollary}[Finite Classical Zorn]
If $P$ is a \emph{classical} probabilistic poset (all values in $\{\pzero, \pone\}$) on a finite nonempty type, then there exists a maximal element.
\end{corollary}

\begin{proof}
By Theorem~\ref{thm:finite-zorn}, there exists an $\varepsilon^*$-near-maximal element $m$.

For classical posets, $\plt(x,y) \in \{\pzero, \pone\}$. If $\varepsilon^* = \pzero$, then $m$ is truly maximal, and we are done.

Suppose for contradiction that $\varepsilon^* = \pone$. Then for every $a_i$, there exists $a_j$ with $\plt(a_i, a_j) = \pone$. Construct a chain: pick any $a_1$, then $a_2$ with $a_1 \plt a_2$, then $a_3$ with $a_2 \plt a_3$, etc. Since $\alpha$ is finite, this chain must eventually repeat: $a_k = a_\ell$ for some $k < \ell$.

By transitivity of $\plt$, we have $a_k \plt a_\ell = a_k$. But $\plt$ is irreflexive (proven in Paper 4: $\plt(x,x) = \pzero$ for all $x$). Contradiction.

Thus $\varepsilon^* = \pzero$ and $m$ is maximal.
\end{proof}

\subsection{Examples}

\begin{example}[Three-Element Chain]
Let $\alpha = \{a, b, c\}$ with classical ordering $a < b < c$. Then:
\begin{itemize}
    \item $\delta(a) = \max(\plt(a,b), \plt(a,c)) = \max(\pone, \pone) = \pone$
    \item $\delta(b) = \max(\plt(b,a), \plt(b,c)) = \max(\pzero, \pone) = \pone$
    \item $\delta(c) = \max(\plt(c,a), \plt(c,b)) = \max(\pzero, \pzero) = \pzero$
\end{itemize}
Thus $\varepsilon^* = \min(\pone, \pone, \pzero) = \pzero$, and $c$ is the (unique) maximal element.
\end{example}

\begin{example}[Probabilistic Triangle]
Let $\alpha = \{a, b, c\}$ with:
\[
P.\mathsf{le}(a,b) = 0.7, \quad P.\mathsf{le}(b,c) = 0.6, \quad P.\mathsf{le}(c,a) = 0.5
\]
and symmetrically for reverse directions (summing to 1). This is a ``rock-paper-scissors'' structure with no clear winner. Computing $\delta$ for each element shows $\varepsilon^* > 0$---there is no truly maximal element, only near-maximal ones.
\end{example}

\subsection{What the Finite Case Establishes}

Theorem~\ref{thm:finite-zorn} shows:

\begin{enumerate}
    \item \textbf{Finite Zorn is choice-free}: No AC, DC, or CC is needed.
    \item \textbf{Near-maximality is definable}: Given a finite $\alpha$ with decidable ordering, $\varepsilon^*$ and a witness are definable without additional axioms.
    \item \textbf{Foundation for infinite case}: The infinite case will use finite approximations.
\end{enumerate}

The key insight: for finite sets, we never need to \emph{select}---we can \emph{compute} by enumeration.

\section{Part 3: Projective Limits and Infinite Zorn}
\label{sec:projective-limits}
% ============================================================================

We sketch how probabilistic Zorn for infinite types follows from the finite case via projective limits. This section presents the strategy; full formalization is future work.

\subsection{The Strategy}

\textbf{Key insight}: We construct \emph{distributions} on near-maximal elements, not individual near-maximal elements. Distributions can be built as limits of finite approximations.

\begin{enumerate}
    \item For each finite subset $F \subset \alpha$, apply finite Zorn to get a distribution $D_F$ concentrated on near-maximals of $F$.
    \item These distributions form a \emph{projective system}: if $F \subset F'$, then $D_{F'}$ projects to $D_F$.
    \item The \emph{Kolmogorov extension theorem} gives a limit distribution $D$ on $\alpha$.
    \item We argue that $D$ is concentrated on near-maximals of $\alpha$.
\end{enumerate}

\subsection{Finite Approximations}

\begin{definition}[Finite Restriction]
For finite $F \subset \alpha$, define the restricted poset $P|_F$ with:
\[
(P|_F).\mathsf{le}(x,y) := P.\mathsf{le}(x,y) \quad \text{for } x, y \in F
\]
\end{definition}

\begin{definition}[Near-Maximal in $F$]
Element $x \in F$ is \emph{$\varepsilon$-near-maximal in $F$} if:
\[
\forall y \in F,\, \plt|_F(x,y) \ple \varepsilon
\]
\end{definition}

\begin{lemma}
If $x$ is $\varepsilon$-near-maximal in $F' \supset F$ and $x \in F$, then $x$ is $\varepsilon$-near-maximal in $F$.
\end{lemma}

\begin{proof}
Near-maximality in $F'$ means $\plt(x,y) \leq \varepsilon$ for all $y \in F'$. Since $F \subset F'$, this holds for all $y \in F$.
\end{proof}

\subsection{The Projective System}

For each finite $F$, let:
\[
M_F^\varepsilon := \{ x \in F : x \text{ is } \varepsilon\text{-near-maximal in } F \}
\]

\begin{definition}[Distribution on $F$]
Define $D_F : F \to \Prob$ as the uniform distribution on $M_F^{\varepsilon_F}$:
\[
D_F(x) := \begin{cases}
    1/|M_F^{\varepsilon_F}| & \text{if } x \in M_F^{\varepsilon_F} \\
    \pzero & \text{otherwise}
\end{cases}
\]
where $\varepsilon_F$ is the optimal bound from Theorem~\ref{thm:finite-zorn}.
\end{definition}

\begin{conjecture}[Projective Property]
For $F \subset F'$, the distributions satisfy a consistency condition enabling projective limits.
\end{conjecture}

\begin{remark}
The precise consistency condition is not yet formulated. The difficulty: $\varepsilon_{F'}$ may differ from $\varepsilon_F$, and near-maximality is relative to the ambient set. A fixed $\varepsilon$ across all $F$ would simplify this but may not exist.
\end{remark}

\subsection{Kolmogorov Extension}

\begin{theorem}[Kolmogorov Extension, Classical]
Let $\{D_F\}_{F \text{ finite}}$ be a projective system of probability distributions. If the system is consistent (projections agree), there exists a distribution $D$ on $\alpha$ such that $D|_F = D_F$ for all finite $F$.
\end{theorem}

\begin{remark}
The classical proof uses:
\begin{enumerate}
    \item Carath\'eodory extension for measures
    \item Compactness arguments for consistency
\end{enumerate}
In our setting, we work with distributions directly (functions to $\Prob$), which may simplify some aspects.
\end{remark}

\subsubsection*{Measure-Theoretic vs.\ Synthetic Limits}

\textbf{Gap}: Classical Kolmogorov applies to probability \emph{measures} on $\sigma$-algebras. Our ``distributions'' are functions $\alpha \to \Prob$, not measures.

Two paths forward:
\begin{enumerate}
    \item \textbf{Measure-theoretic}: Construct actual measures, apply classical Kolmogorov, then extract our distributions. Requires measure theory infrastructure.
    \item \textbf{Synthetic}: Develop projective limits directly for $\Prob$-valued functions. This would be new theory, more aligned with our synthetic approach.
\end{enumerate}

This is a major open problem for Paper 5.

\subsection{The Conjecture}

\begin{conjecture}[Infinite Probabilistic Zorn]
\label{conj:infinite-zorn}
Let $\alpha$ be a nonempty type with chain-complete probabilistic poset structure $P$. For any $\varepsilon > \pzero$, there exists a distribution $D$ on $\alpha$ concentrated on $\varepsilon$-near-maximal elements.
\end{conjecture}

\begin{proof}[Proof Strategy]
\begin{enumerate}
    \item Construct the projective system $\{D_F\}$ as above.
    \item Apply Kolmogorov extension (or synthetic analog) to get limit $D$.
    \item Show: if $D(x) \neq \pzero$, then $x$ is $\varepsilon$-near-maximal.
\end{enumerate}

\textbf{Step 3 argument} (contrapositive): Suppose $x$ is \emph{not} $\varepsilon$-near-maximal in $\alpha$. Then there exists $y \in \alpha$ with $\plt(x,y) > \varepsilon$. Consider any finite $F$ containing both $x$ and $y$. In $F$, we have $\plt|_F(x,y) = \plt(x,y) > \varepsilon$, so $x$ is not $\varepsilon$-near-maximal in $F$. Thus $D_F(x) = \pzero$ for all such $F$. By the limit construction, $D(x) = \pzero$.
\end{proof}

\subsection{Why No Choice?}

\textbf{Classical Zorn} requires selecting a maximal element from chains---this is where AC enters.

\textbf{Probabilistic Zorn} constructs a distribution:
\begin{itemize}
    \item Finite approximations: no choice (enumeration suffices)
    \item Projective system: canonical construction (no selection)
    \item Kolmogorov extension: limit construction (not selection)
\end{itemize}

We never select a point from an infinite set. We construct a measure as a limit of finite measures. This is why the probabilistic version should be choice-free.

\subsection{Gaps to Fill}

\begin{enumerate}
    \item \textbf{Consistency of projective system}: Show $D_{F'}|_F = D_F$ (or appropriate variant).
    \item \textbf{Kolmogorov in our framework}: Formalize for $\Prob$-valued distributions (synthetic analog).
    \item \textbf{Near-maximality at limit}: The contrapositive argument in the proof strategy needs the limit construction to respect the property.
    \item \textbf{Chain-completeness role}: See remark below.
\end{enumerate}

\begin{remark}[Role of Chain-Completeness]
Chain-completeness ensures every chain has an upper bound. In classical Zorn, this prevents infinite ascending chains from ``escaping to infinity.'' In our probabilistic setting, the precise role is unclear. One possibility: chain-completeness ensures the projective system has a well-defined limit (chains in the poset of finite subsets have upper bounds). Investigation ongoing.
\end{remark}

Resolving these gaps would turn Conjecture~\ref{conj:infinite-zorn} into a theorem and allow us to remove the \texttt{prob\_zorn} axiom.
