\title{ProbTT: Type Theory with Primitive Weights\\[0.5em]
\large Weights from a De Morgan Algebra}

\author{Probabilistic Foundations Project}
\date{\today}

\begin{document}

\maketitle

\begin{abstract}
We define a type theory where judgments carry \emph{weights} from a De Morgan algebra. The algebra has multiplication, complement, zero, one, and order---with disjunction derived via De Morgan duality. When $\W = \{\zero, \one\}$, we recover Martin-L\"of Type Theory. When $\W = {[0,1]}$, we get probabilistic typing with conditioning via the chain rule (not division). The key insight: types and logic \emph{emerge} from weight structure. We add conditioning as a primitive, distinguishing this from purely algebraic approaches.
\end{abstract}

%=============================================================================
\section{The Weight Algebra}
%=============================================================================

\subsection{Product De Morgan Structure}

We axiomatize a minimal structure: multiplication and complement. Disjunction is derived.

\begin{definition}[Weight Algebra]
$\W$ is a type with:
\begin{itemize}
    \item Constants: $\one : \W$ (certainty), $\zero : \W$ (impossibility)
    \item Multiplication: $(\cdot) : \W \to \W \to \W$ (conjunction)
    \item Complement: $\negw : \W \to \W$ (negation)
    \item Order: $(\leq) : \W \to \W \to \Type$
\end{itemize}
\end{definition}

\begin{definition}[Multiplication Axioms]
\begin{align}
w \cdot \one &= w \tag{unit-right} \\
\one \cdot w &= w \tag{unit-left} \\
w \cdot \zero &= \zero \tag{annihilate-right} \\
\zero \cdot w &= \zero \tag{annihilate-left} \\
(u \cdot v) \cdot w &= u \cdot (v \cdot w) \tag{assoc} \\
u \cdot v &= v \cdot u \tag{comm}
\end{align}
\end{definition}

\begin{definition}[Complement Axioms]
\begin{align}
\negw \zero &= \one \tag{neg-zero} \\
\negw \one &= \zero \tag{neg-one} \\
\negw \negw w &= w \tag{involution}
\end{align}
\end{definition}

\begin{definition}[Order Axioms]
\begin{align}
w &\leq w \tag{refl} \\
u \leq v \land v \leq w &\Rightarrow u \leq w \tag{trans} \\
u \leq v \land v \leq u &\Rightarrow u = v \tag{antisym} \\
\zero &\leq w \tag{zero-least} \\
w &\leq \one \tag{one-greatest}
\end{align}
\end{definition}

\subsection{Derived Operations}

\begin{definition}[De Morgan Disjunction]
\[
w \vee v := \negw(\negw w \cdot \negw v)
\]
\end{definition}

When $\W = {[0,1]}$ with $\negw w = 1 - w$, we get:
\[
w \vee v = 1 - (1-w)(1-v) = w + v - wv
\]
This is the \textbf{probabilistic sum} (inclusion-exclusion for independent events).

\begin{theorem}[De Morgan Laws]
\begin{align}
\negw(w \cdot v) &= \negw w \vee \negw v \\
\negw(w \vee v) &= \negw w \cdot \negw v
\end{align}
\end{theorem}

\begin{proof}
Direct calculation from definitions.
\end{proof}

\subsection{What We Have vs What We Don't}

\begin{center}
\begin{tabular}{l|l}
\textbf{Primitive} & \textbf{Derived} \\
\hline
$\cdot$ (multiplication/AND) & $\vee$ (De Morgan OR) \\
$\negw$ (complement/NOT) & $\to$ (implication, if needed) \\
$\zero, \one$ (constants) & \\
$\leq$ (order) &
\end{tabular}
\end{center}

\textbf{NOT primitive}: addition $+$, subtraction $-$, division $/$.

\subsection{Instances}

\begin{example}[Boolean: $\W = \{\zero, \one\}$]
$\one \cdot \one = \one$, else $\zero$. $\negw \zero = \one$, $\negw \one = \zero$.

This is classical Boolean logic.
\end{example}

\begin{example}[Probability: $\W = {[0,1]}$]
$w \cdot v = w \times v$ (ordinary multiplication). $\negw w = 1 - w$.

This is product fuzzy logic / probabilistic reasoning.
\end{example}

\begin{example}[Three-valued: $\W = \{\zero, ?, \one\}$]
$? \cdot ? = ?$, $\negw ? = ?$.

Kleene's strong three-valued logic.
\end{example}

\section{Judgments}
%=============================================================================

\begin{definition}[Judgment Forms]
\begin{enumerate}
    \item $\Gamma \ctx$ \hfill (valid context)
    \item $\Gamma \vdash A \type$ \hfill (valid type)
    \item $\Gamma \vdash a : A \at{w}$ \hfill (weighted typing)
    \item $\Gamma \vdash a \equiv b : A \at{w}$ \hfill (weighted equality)
\end{enumerate}
\end{definition}

\begin{remark}[Interpretation]
$\Gamma \vdash a : A \at{w}$ means: term $a$ inhabits type $A$ with weight $w$.
\begin{itemize}
    \item $w = \one$: $a$ definitely inhabits $A$
    \item $w = \zero$: $a$ does not inhabit $A$ (vacuously true)
    \item $\zero < w < \one$: $a$ partially/probably inhabits $A$
\end{itemize}
\end{remark}

\end{gather}

\section{Type Formers}
%=============================================================================

\subsection{Functions (Dependent)}

\begin{gather}
\infer[\textsc{$\Pi$-Form}]
  {\Gamma \vdash (x : A) \to B \type}
  {\Gamma \vdash A \type \quad \Gamma, x : A \vdash B \type}
\\[1em]
\infer[\textsc{$\Pi$-Intro}]
  {\Gamma \vdash \lambda x. b : (x : A) \to B \at{w}}
  {\Gamma, x : A \vdash b : B \at{w}}
\\[1em]
\infer[\textsc{$\Pi$-Elim}]
  {\Gamma \vdash f\,a : B[a/x] \at{w \cdot v}}
  {\Gamma \vdash f : (x : A) \to B \at{w} \quad \Gamma \vdash a : A \at{v}}
\end{gather}

\textbf{Key rule}: weights multiply in elimination.

\subsection{Pairs (Dependent)}

\begin{gather}
\infer[\textsc{$\Sigma$-Form}]
  {\Gamma \vdash \Sigma(x : A). B \type}
  {\Gamma \vdash A \type \quad \Gamma, x : A \vdash B \type}
\\[1em]
\infer[\textsc{$\Sigma$-Intro}]
  {\Gamma \vdash (a, b) : \Sigma(x : A). B \at{w \cdot v}}
  {\Gamma \vdash a : A \at{w} \quad \Gamma \vdash b : B[a/x] \at{v}}
\\[1em]
\infer[\textsc{$\Sigma$-Elim$_1$}]
  {\Gamma \vdash \pi_1\,t : A \at{w}}
  {\Gamma \vdash t : \Sigma(x : A). B \at{w}}
\qquad
\infer[\textsc{$\Sigma$-Elim$_2$}]
  {\Gamma \vdash \pi_2\,t : B[\pi_1 t / x] \at{w}}
  {\Gamma \vdash t : \Sigma(x : A). B \at{w}}
\end{gather}

Pair weight = product of component weights (joint probability).

\subsection{Sums (Coproducts)}

\begin{gather}
\infer[\textsc{$+$-Form}]
  {\Gamma \vdash A + B \type}
  {\Gamma \vdash A \type \quad \Gamma \vdash B \type}
\\[1em]
\infer[\textsc{$+$-Intro$_L$}]
  {\Gamma \vdash \mathsf{inl}\,a : A + B \at{w}}
  {\Gamma \vdash a : A \at{w}}
\qquad
\infer[\textsc{$+$-Intro$_R$}]
  {\Gamma \vdash \mathsf{inr}\,b : A + B \at{w}}
  {\Gamma \vdash b : B \at{w}}
\\[1em]
\infer[\textsc{$+$-Elim}]
  {\Gamma \vdash \mathsf{case}\,e\,\mathsf{of}\,\mathsf{inl}\,x \Rightarrow l \mid \mathsf{inr}\,y \Rightarrow r : C \at{w \cdot v}}
  {\Gamma \vdash e : A + B \at{w} \quad \Gamma, x : A \vdash l : C \at{v} \quad \Gamma, y : B \vdash r : C \at{v}}
\end{gather}

\textbf{Key rule}: weights multiply in sum elimination. The scrutinee weight $w$ combines with the branch weight $v$ to give joint weight $w \cdot v$.

\subsection{Empty and Unit}

\begin{gather}
\infer[\textsc{$\mathbf{0}$-Form}]
  {\Gamma \vdash \mathbf{0} \type}
  {\Gamma \ctx}
\qquad
\infer[\textsc{$\mathbf{0}$-Elim}]
  {\Gamma \vdash \mathsf{abort}\,e : A \at{w}}
  {\Gamma \vdash e : \mathbf{0} \at{w}}
\\[1em]
\infer[\textsc{$\mathbf{1}$-Form}]
  {\Gamma \vdash \mathbf{1} \type}
  {\Gamma \ctx}
\qquad
\infer[\textsc{$\mathbf{1}$-Intro}]
  {\Gamma \vdash \star : \mathbf{1} \at{\one}}
  {\Gamma \ctx}
\end{gather}

\subsection{Identity}

\begin{gather}
\infer[\textsc{Id-Form}]
  {\Gamma \vdash \mathsf{Id}_A(a, b) \type}
  {\Gamma \vdash a : A \at{\one} \quad \Gamma \vdash b : A \at{\one}}
\\[1em]
\infer[\textsc{Id-Intro}]
  {\Gamma \vdash \mathsf{refl} : \mathsf{Id}_A(a, a) \at{w}}
  {\Gamma \vdash a : A \at{w}}
\\[1em]
\infer[\textsc{Id-Elim}]
  {\Gamma \vdash J(M, d, p) : M[b, p] \at{w \cdot v}}
  {\Gamma \vdash p : \mathsf{Id}_A(a, b) \at{w} \quad \Gamma \vdash d : M[a, \mathsf{refl}] \at{v}}
\end{gather}

The J eliminator transports along an identity proof. Weights multiply: the proof weight $w$ combines with the base case weight $v$.

\section{Computation Rules}
%=============================================================================

The computation (beta) rules reduce eliminator-intro pairs. In each case, weight multiplication is preserved.

\subsection{$\Pi$-Computation}
\[
(\lambda x. b)\,a \equiv b[a/x] : B[a/x] \at{w \cdot v}
\]
when $\Gamma, x : A \vdash b : B \at{w}$ and $\Gamma \vdash a : A \at{v}$.

\subsection{$\Sigma$-Computation}
\begin{align}
\pi_1(a, b) &\equiv a : A \at{w \cdot v} \\
\pi_2(a, b) &\equiv b : B[a/x] \at{w \cdot v}
\end{align}
when $\Gamma \vdash a : A \at{w}$ and $\Gamma \vdash b : B[a/x] \at{v}$.

\subsection{$+$-Computation}
\begin{align}
\mathsf{case}\,(\mathsf{inl}\,a)\,l\,r &\equiv l[a/x] : C \at{w \cdot v} \\
\mathsf{case}\,(\mathsf{inr}\,b)\,l\,r &\equiv r[b/y] : C \at{w \cdot v}
\end{align}

\subsection{Id-Computation}
\[
J(M, d, \mathsf{refl}) \equiv d : M[a, \mathsf{refl}] \at{w \cdot v}
\]
when $\Gamma \vdash a : A \at{w}$ and $\Gamma \vdash d : M[a, \mathsf{refl}] \at{v}$.

\section{Negation}
%=============================================================================

There are two approaches to negation:

\subsection{Type-Theoretic Negation}

\[
\neg A := A \to \mathbf{0}
\]

A proof of $\neg A$ is a function from $A$ to the empty type.

\subsection{Weight Negation}

For a term $a : A \at{w}$, we can consider $\negw w$ as the ``complementary weight.''

In $\W = {[0,1]}$: if $P(A) = w$, then $P(\neg A) = 1 - w = \negw w$.

\begin{remark}[Relationship]
Type-theoretic negation ($A \to \mathbf{0}$) corresponds to weight negation in the following sense: if $a : A \at{w}$ with $w > \zero$, then there is no $f : A \to \mathbf{0} \at{\one}$.
\end{remark}

\section{Conditioning (Not Implication)}
%=============================================================================

\subsection{Why Not Material Implication}

Material implication $A \to B := \negw A \vee B$ has paradoxes:
\begin{itemize}
    \item $\mathsf{False} \to \mathsf{Anything} = \mathsf{True}$ (ex falso)
    \item $\mathsf{Anything} \to \mathsf{True} = \mathsf{True}$
\end{itemize}

We use \textbf{conditioning} instead of implication.

\subsection{Chain Rule}

The fundamental equation is:
\[
P(A, B) = P(A \mid B) \cdot P(B) = P(B \mid A) \cdot P(A)
\]

This is \textbf{multiplicative}---no division needed!

\begin{gather}
\infer[\textsc{Chain}]
  {\Gamma \vdash (a, b) : A \times B \at{u \cdot v}}
  {\Gamma \vdash a : A \at{u} \quad \Gamma \vdash b \mid a : B \at{v}}
\end{gather}

The conditional $b \mid a$ is \textbf{primitive}. We don't compute it by division; we assert it exists when the chain rule holds.

\subsection{Conditioning vs Implication}

\begin{center}
\begin{tabular}{l|l|l}
& \textbf{Implication} $A \to B$ & \textbf{Conditioning} $A \mid B$ \\
\hline
Formula & $\negw A \vee B$ & $P(A,B) / P(B)$ \\
Computable from & $A$ and $B$ alone & Joint $P(A,B)$ \\
False antecedent & ``True'' (lies) & Indeterminate (honest) \\
In logic & Yes & No \\
In probability & No & Yes \\
\end{tabular}
\end{center}

\subsection{Graded Ex Falso}

What about $P(A \mid B)$ when $P(B) = 0$?

In classical logic: $\bot \to A$ is always true (ex falso quodlibet).

In ProbTT: when $b : B \at{\zero}$:
\[
(a, b) : A \times B \at{\zero \cdot w} = \at{\zero}
\]
The conditional $a \mid b$ can have \emph{any} weight---the joint is already zero.

\textbf{Key insight}: Ex falso is the $w = 0$ limit of a continuous phenomenon:
\begin{itemize}
    \item $w = \one$: Evidence fully constrains conclusions
    \item $w = \epsilon$: Evidence weakly constrains conclusions
    \item $w = \zero$: Evidence imposes no constraint (ex falso)
\end{itemize}

Classical logic has only the endpoints; ProbTT has the entire spectrum.

\section{MLTT as the $\{\zero, \one\}$ Case}
%=============================================================================

\begin{theorem}[MLTT Embedding]
When $\W = \{\zero, \one\}$, ProbTT reduces to MLTT:
\begin{itemize}
    \item $\Gamma \vdash a : A \at{\one}$ becomes $\Gamma \vdash a : A$
    \item $\Gamma \vdash a : A \at{\zero}$ is vacuously satisfied
    \item Weight operations: $\one \cdot \one = \one$, else $\zero$
    \item Complement: $\negw\zero = \one$, $\negw\one = \zero$
    \item De Morgan OR: $w \vee v = \negw(\negw w \cdot \negw v)$ gives Boolean OR
\end{itemize}
\end{theorem}

\begin{proof}
By inspection. All weight operations compute to Boolean.
\end{proof}

\subsection{Side-by-Side Comparison}

\begin{center}
\begin{tabular}{l|l|l}
\textbf{Rule} & \textbf{MLTT} & \textbf{ProbTT} \\
\hline
Variable & $x : A \vdash x : A$ & $x : A \vdash x : A \at{\one}$ \\
Application & $f\,a : B[a/x]$ & $f\,a : B[a/x] \at{w \cdot v}$ \\
Pair & $(a, b) : \Sigma A\,B$ & $(a, b) : \Sigma A\,B \at{w \cdot v}$ \\
Projection & $\pi_1 t : A$ & $\pi_1 t : A \at{w}$ \\
Case & $\mathsf{case}\,e\,l\,r : C$ & $\mathsf{case}\,e\,l\,r : C \at{w \cdot v}$ \\
\end{tabular}
\end{center}

\textbf{Key observation}: In MLTT, all judgments are implicitly at weight $\one$. Since $\one \cdot \one = \one$, weight multiplication is invisible. ProbTT makes this structure explicit and generalizes it to arbitrary weights.

\section{Contrast with Cost-Based Graded Types}
%=============================================================================

Graded type systems like Granule \citep{orchard2019quantitative} and quantitative type theory \citep{atkey2018syntax} use grades to track \emph{resource usage}. ProbTT uses weights to track \emph{inhabitation certainty}. These are philosophically distinct:

\begin{center}
\begin{tabular}{l|l|l}
& \textbf{Cost-Based (Granule)} & \textbf{ProbTT} \\
\hline
Algebraic structure & Semiring $(0, 1, +, \cdot)$ & De Morgan $(\zero, \one, \cdot, \negw)$ \\
Addition & Primitive (branch costs add) & Derived via De Morgan \\
Philosophy & Decorate MLTT with grades & Logic \emph{emerges} from grades \\
Implication & Material implication exists & Implication \emph{rejected} \\
Interpretation & Costs accumulate & Weights condition/constrain \\
MLTT relationship & MLTT is axiomatic & MLTT is \emph{derivative} \\
\end{tabular}
\end{center}

\textbf{Key distinction}: Cost-based systems ask ``how much resource does this computation use?'' ProbTT asks ``how certain are we that this type is inhabited?''

In cost-based grading:
\begin{itemize}
    \item $f : A \to B @ 2$ means ``$f$ uses 2 units of resource''
    \item Costs add at branches: if one branch costs 3 and another costs 5, total cost is at most 5
    \item The semiring axiom $0 + x = x$ reflects ``zero cost contributes nothing''
\end{itemize}

In ProbTT:
\begin{itemize}
    \item $f : A \to B \at{0.9}$ means ``$f$ produces $B$ with 90\% certainty''
    \item Weights multiply in composition: chaining 90\% and 80\% gives 72\%
    \item The De Morgan axiom $\zero \cdot w = \zero$ reflects ``impossible antecedent gives impossible conclusion''
\end{itemize}

\textbf{One-liner}: Costs measure \emph{computation performed}; weights measure \emph{truth surviving composition}.

%=============================================================================
\section{Related Work: A Unified View}
%=============================================================================

ProbTT synthesizes ideas from several traditions. We review each and explain how ProbTT unifies them.

\subsection{Pavelka-Style Fuzzy Logic}

Pavelka \citep{pavelka1979fuzzy} introduced \emph{graded provability} for \textbf{propositional} fuzzy logic: instead of asking ``is $\varphi$ provable?'', one asks ``to what degree is $\varphi$ provable?''

\begin{definition}[Pavelka Provability Degree]
\[
|\varphi|_T := \sup\{r \in [0,1] \mid T \vdash (\varphi, r)\}
\]
\end{definition}

Pavelka \textbf{proved} a completeness theorem for this propositional system, equating syntactic provability degrees with semantic truth degrees.

H\'ajek \citep{hajek1998metamathematics} extended this to \textbf{predicate} fuzzy logic in his \emph{Metamathematics of Fuzzy Logic}. His completeness theorems apply to first-order fuzzy logic, not type theory.

\textbf{Key distinction}: Pavelka/H\'ajek work in propositional and first-order logic respectively, and they \textbf{prove} their completeness results. ProbTT works in dependent type theory, which is a different formal system. The completeness question for ProbTT (Section 11, Open Problem 7) remains open.

\textbf{What ProbTT adapts from Pavelka/H\'ajek}:
\begin{itemize}
    \item The concept of graded provability (weights on judgments)
    \item The idea that degrees should form an algebra
\end{itemize}

\textbf{What ProbTT adds}:
\begin{itemize}
    \item Dependent types ($\Pi$, $\Sigma$, $\mathsf{Id}$)
    \item Weights on \emph{judgments}, not just propositions
    \item Conditioning as primitive (not residuated implication)
\end{itemize}

\textbf{What ProbTT postulates} (analogous to what Pavelka/H\'ajek prove):
\begin{itemize}
    \item That G\"odel sentences attain weight $\frac{1}{2}$ (we postulate the fixed point; they prove completeness)
\end{itemize}

\subsection{Self-Reference in Fuzzy Logic}

H\'ajek, Paris, and Shepherdson \citep{hajek2000liar} analyzed the liar paradox in fuzzy logic:

\begin{theorem}[H\'ajek-Paris-Shepherdson]
In {\L}ukasiewicz logic with a truth predicate $T$, the liar sentence $\lambda$ satisfying $\lambda \leftrightarrow \neg T(\ulcorner \lambda \urcorner)$ has truth value $\frac{1}{2}$.
\end{theorem}

\textbf{Key point}: H\'ajek-Paris-Shepherdson \textbf{prove} this result by carefully constructing the truth predicate and showing the fixed point exists. Their proof uses the diagonal lemma and properties of {\L}ukasiewicz logic.

\textbf{ProbTT's version}: We \textbf{postulate} that the G\"odel sentence $G \leftrightarrow \neg\mathsf{Prov}_\one(G)$ exists and has weight satisfying $w = \negw w$, hence $w = \frac{1}{2}$. This uses the same fixed-point \emph{phenomenon}, but:
\begin{itemize}
    \item H\'ajek-Paris-Shepherdson work with \emph{truth} in {\L}ukasiewicz logic
    \item ProbTT works with \emph{provability weights} in type theory
    \item They \textbf{construct} $\lambda$; we \textbf{postulate} $G$'s existence
\end{itemize}

A full treatment would require proving the diagonal lemma for ProbTT and constructing $G$ explicitly.

\subsection{Logics of Formal Inconsistency (LFIs)}

Da Costa \citep{dacosta1974paraconsistent} and Carnielli-Coniglio \citep{carnielli2016paraconsistent} developed logics where contradiction doesn't explode:

\begin{definition}[Gentle Explosion]
\[
A, \neg A, \circ A \vdash B \quad \text{but} \quad A, \neg A \not\vdash B
\]
where $\circ A$ means ``$A$ is consistent.''
\end{definition}

LFIs have a well-developed proof theory with the consistency operator $\circ$ as \textbf{primitive syntax}. The operator gates explosion: you need $\circ A$ (assertion that $A$ is consistent) to derive arbitrary conclusions from $A \land \neg A$.

\textbf{ProbTT's approach}: ProbTT has no consistency operator and no explosion rule. From $a : A \at{w}$ and $b : \neg A \at{v}$, we simply cannot derive arbitrary $B \at{\one}$---there is no rule that would produce this. This is not ``gated explosion'' but \emph{absence of explosion}.

\begin{center}
\begin{tabular}{l|l|l}
& \textbf{LFIs} & \textbf{ProbTT} \\
\hline
Explosion & Gated by $\circ$ operator & Absent (no explosion rule) \\
Formal device & Primitive unary connective & Graded judgments \\
Gradation & Binary (consistent/not) & Continuous $[0,1]$ \\
Proof theory & Well-developed & Postulated analogs \\
\end{tabular}
\end{center}

The comparison is suggestive but limited: LFIs and ProbTT address paraconsistency through different mechanisms.

\subsection{T-Norm Fuzzy Logics}

Most fuzzy logics use \emph{t-norms} for conjunction \citep{klement2000triangular}:

\begin{definition}[T-norm]
A t-norm $* : [0,1]^2 \to [0,1]$ satisfies:
\begin{enumerate}
    \item $x * 1 = x$ (identity)
    \item $x * y = y * x$ (commutativity)
    \item $(x * y) * z = x * (y * z)$ (associativity)
    \item $x \leq y \Rightarrow x * z \leq y * z$ (monotonicity)
\end{enumerate}
\end{definition}

Common t-norms: product ($x \cdot y$), minimum ($\min(x,y)$), Łukasiewicz ($\max(0, x+y-1)$).

\textbf{ProbTT uses product}: $w \cdot v$ is ordinary multiplication. This matches probabilistic conjunction (independent events) and gives the chain rule.

\textbf{Key difference}: Standard fuzzy logics include a \emph{residuum} (implication) defined by:
\[
x \to y := \sup\{z \mid x * z \leq y\}
\]

ProbTT \emph{rejects} this implication in favor of conditioning. The residuum has material implication paradoxes; conditioning doesn't.

\subsection{Graded Type Theory (Granule)}

Orchard et al. \citep{orchard2019quantitative} and Atkey \citep{atkey2018syntax} developed graded type systems for resource tracking:

\begin{center}
\begin{tabular}{l|l|l}
& \textbf{Granule/QTT} & \textbf{ProbTT} \\
\hline
Algebraic structure & Semiring $(0,1,+,\cdot)$ & De Morgan $(\zero,\one,\cdot,\negw)$ \\
Addition & Primitive (costs add) & Derived (De Morgan) \\
Interpretation & Resource usage & Inhabitation certainty \\
\end{tabular}
\end{center}

\textbf{Technical distinction}: Granule's semiring includes addition as primitive, enabling resource branching: ``use resource $r$ in one branch OR resource $s$ in another.'' ProbTT derives addition via De Morgan duality, which limits expressiveness for branching scenarios.

\textbf{Philosophical distinction}: Both systems treat grades as primitive. The difference is interpretive:
\begin{itemize}
    \item Granule: ``how much resource does this computation use?''
    \item ProbTT: ``how certain is this type inhabited?''
\end{itemize}

In Granule, $f : A \to B @ 2$ means ``$f$ uses 2 units.'' In ProbTT, $f : A \to B \at{0.9}$ means ``$f$ produces $B$ with 90\% certainty.''

\textbf{Note}: The claim that MLTT ``emerges'' from ProbTT while Granule ``decorates'' MLTT is a matter of presentation. Both can be viewed as parameterized systems that specialize to MLTT. Granule's semiring can also be instantiated at $\{0,1\}$.

\subsection{De Morgan Algebras}

Our weight algebra is a \emph{De Morgan algebra} \citep{kalman1958lattices}:

\begin{definition}[De Morgan Algebra]
$(L, \land, \lor, \neg, 0, 1)$ with:
\begin{itemize}
    \item $(L, \land, \lor, 0, 1)$ is a bounded distributive lattice
    \item $\neg\neg x = x$ (involution)
    \item $\neg(x \land y) = \neg x \lor \neg y$ (De Morgan)
\end{itemize}
\end{definition}

ProbTT uses a \emph{product De Morgan algebra}: conjunction is multiplication, disjunction derived via De Morgan. When $L = [0,1]$ with $\neg w = 1-w$:
\begin{align}
w \land v &= w \cdot v \\
w \lor v &= \negw(\negw w \cdot \negw v) = w + v - wv
\end{align}

\subsection{Markov Categories}

Fritz \citep{fritz2020synthetic} developed Markov categories for probabilistic reasoning:

\begin{definition}[Markov Category]
A symmetric monoidal category with:
\begin{itemize}
    \item Copy: $\Delta_A : A \to A \otimes A$
    \item Delete: $!_A : A \to I$
    \item Naturality conditions for stochastic maps
\end{itemize}
\end{definition}

ProbTT's categorical semantics is a \emph{weighted Markov category}: morphisms carry weights, composition multiplies weights.

\subsection{Synthesis: What ProbTT Unifies}

\begin{center}
\begin{tabular}{l|l}
\textbf{From} & \textbf{ProbTT takes} \\
\hline
Pavelka/Hájek & Graded provability \\
Hájek-Paris-Shepherdson & Self-reference $\to$ fixed point at $\frac{1}{2}$ \\
LFIs & Non-explosive contradiction \\
T-norm logics & Product conjunction \\
Granule/QTT & Dependent types + grades \\
De Morgan algebras & Minimal algebraic structure \\
Markov categories & Conditioning semantics \\
\end{tabular}
\end{center}

\textbf{What ProbTT combines}: Dependent type theory with conditioning (not implication), using De Morgan algebra for weights. MLTT is the $\{\zero,\one\}$ specialization.

\textbf{Honest assessment of novelty}: The individual components (graded provability, fuzzy fixed points, graded types, De Morgan algebras) are well-established. ProbTT's contribution is combining them with conditioning as primitive and exploring the consequences for incompleteness and choice. Many results are postulated analogs of proven theorems in simpler settings.

\subsection{What ProbTT Rejects}

\begin{enumerate}
    \item \textbf{Implication}: No $A \to B := \negw A \lor B$. Use conditioning instead.
    \item \textbf{Addition}: No primitive $+$. Disjunction derived via De Morgan.
    \item \textbf{Division}: No $P(A|B) = P(A,B)/P(B)$. Use chain rule multiplicatively.
    \item \textbf{MLTT as foundation}: MLTT is the $\{\zero,\one\}$ \emph{limit}, not the starting point.
\end{enumerate}

\begin{definition}[Weighted Category]
A \emph{$\W$-weighted category} has:
\begin{itemize}
    \item Objects (types)
    \item For each $w : \W$, morphisms $\mathrm{Hom}_w(A, B)$
    \item Identity: $\mathrm{id}_A \in \mathrm{Hom}_\one(A, A)$

\end{itemize}
\end{definition}

\section{Graded Incompleteness}
%=============================================================================

G\"odel's incompleteness theorems have a natural formulation in ProbTT. Rather than forcing undecidability, self-referential sentences attain determinate intermediate weights.

\subsection{Graded Provability Predicate}

\begin{definition}[Graded Provability]
For a theory $T$ and proposition $\varphi$, define:
\[
\mathsf{Prov}_w(\ulcorner \varphi \urcorner) :\equiv \text{there exists a derivation of } \varphi \text{ at weight } w
\]
The \emph{provability degree} is:
\[
|\varphi|_T := \sup\{w \mid T \vdash \varphi \at{w}\}
\]
\end{definition}

This follows Pavelka \citep{pavelka1979fuzzy} and Hájek \citep{hajek1998metamathematics}, who developed graded provability for fuzzy logics.

\subsection{The Diagonal Lemma}

The diagonal lemma is purely syntactic and holds for any sufficiently expressive language:

\begin{lemma}[Diagonal Lemma]
For any predicate $P(x)$, there exists a sentence $G$ such that:
\[
T \vdash G \leftrightarrow P(\ulcorner G \urcorner)
\]
\end{lemma}

In ProbTT, this becomes: for any weight-valued predicate $P$, there exists $G$ such that the weight of $G$ equals $P(\ulcorner G \urcorner)$.

\subsection{The G\"odel Sentence in ProbTT}

\begin{definition}[G\"odel Sentence]
Let $G$ be the sentence asserting ``$G$ is not provable at weight $\one$'':
\[
G \leftrightarrow \neg \mathsf{Prov}_\one(\ulcorner G \urcorner)
\]
\end{definition}

\begin{theorem}[Graded Incompleteness]
If $G$ has weight $w$ in ProbTT, then $w$ satisfies $w = \negw w$, hence $w = \frac{1}{2}$.
\end{theorem}

\begin{proof}
By the definition of $G$: if $G$ has weight $w$, then $G$'s content (that $G$ is not provable at weight $\one$) should also have weight $w$. But $G$'s claim is essentially the negation of provability. In the fixed-point analysis:
\begin{itemize}
    \item If $w = \one$, then $G$ asserts $\neg\mathsf{Prov}_\one(G)$, but $G$ itself is provable at $\one$---contradiction.
    \item If $w = \zero$, then $\neg G$ has weight $\one$, meaning $G$ is provable at $\one$---but we assumed $w = \zero$.
\end{itemize}
The consistent resolution: $w = \negw w$. With $\negw w = 1 - w$:
\[
w = 1 - w \quad \Rightarrow \quad w = \frac{1}{2}
\]
\end{proof}

\begin{remark}[Comparison with Classical Incompleteness]
\begin{center}
\begin{tabular}{l|l|l}
& \textbf{Classical} & \textbf{ProbTT} \\
\hline
$G$'s status & Undecidable & Weight $\frac{1}{2}$ \\
$w = \negw w$ in $\{\zero, \one\}$ & No solution & N/A \\
$w = \negw w$ in $[0,1]$ & N/A & $w = \frac{1}{2}$ \\
Information & Binary gap & Quantified uncertainty \\
\end{tabular}
\end{center}
\end{remark}

The key insight: classical logic's restriction to $\{\zero, \one\}$ forces $w = \negw w$ to have no solution, manifesting as undecidability. ProbTT's continuous weights allow the fixed point to exist at $\frac{1}{2}$.

\subsection{Literature Context}

The fixed point $w = \negw w \Rightarrow w = \frac{1}{2}$ is well-studied in fuzzy logic:

\begin{itemize}
    \item Hájek, Paris, Shepherdson, ``The Liar Paradox and Fuzzy Logic'' (JSL 2000) \citep{hajek2000liar}
    \item Restall on Łukasiewicz logic and self-reference
\end{itemize}

ProbTT's contribution is embedding this in dependent type theory with graded judgments.

\section{Graded Self-Consistency}
%=============================================================================

G\"odel's Second Incompleteness Theorem states that a consistent theory extending PA cannot prove its own consistency. ProbTT offers a graded alternative.

\subsection{The Classical Obstacle}

\begin{theorem}[G\"odel's Second, Classical]
If $T \supseteq PA$ and $T$ is consistent, then $T \not\vdash \mathsf{Con}(T)$.
\end{theorem}

Here $\mathsf{Con}(T) := \neg\mathsf{Prov}_T(\ulcorner \bot \urcorner)$.

\subsection{Graded Consistency}

\begin{definition}[Graded Consistency]
\[
\mathsf{Con}_w(T) :\equiv \text{``$\bot$ is not provable at weight $\one$''} \at{w}
\]
\end{definition}

\begin{theorem}[Graded Second Incompleteness]
ProbTT can prove $\mathsf{Con}_w(\text{ProbTT})$ for some $w < \one$, but not for $w = \one$.
\end{theorem}

\begin{proof}[Proof sketch]
The G\"odel-2 argument requires:
\begin{enumerate}
    \item If $T \vdash \mathsf{Con}(T) \at{\one}$, then by formalized soundness, $T$ proves ``if $T$ proves $\bot$, then $\bot$''
    \item Combined with the G\"odel sentence, this yields $T \vdash G \at{\one}$
    \item But $G$ asserts its own unprovability at $\one$---contradiction
\end{enumerate}

At weight $w < \one$, step 1 fails: we only have $\mathsf{Con}(T) \at{w}$, which doesn't yield the sharp consequence needed. The self-referential loop stabilizes at a weight below $\one$.
\end{proof}

\subsection{Philosophical Interpretation}

\begin{center}
\begin{tabular}{l|l}
\textbf{Classical} & \textbf{ProbTT} \\
\hline
Cannot prove own consistency & Can prove at $w < \one$ \\
``Unknowable'' & ``Confident but not certain'' \\
Binary epistemic gap & Quantified confidence level \\
\end{tabular}
\end{center}

This is philosophically honest: we \emph{believe} our system is consistent, but absolute certainty is impossible (G\"odel). ProbTT quantifies this epistemic state rather than leaving it as a binary gap.

\subsection{Connection to LFIs}

Logics of Formal Inconsistency \citep{carnielli2016paraconsistent} use an object-level consistency operator $\circ A$ (``$A$ is consistent''). The ``gentle explosion'' principle:
\[
A, \neg A, \circ A \vdash B \quad \text{but} \quad A, \neg A \not\vdash B
\]

ProbTT achieves similar effects through weight: contradiction at low weight doesn't explode to arbitrary conclusions because weight multiplication preserves the low weight.

\section{Graded Choice}
%=============================================================================

The Axiom of Choice has a natural graded formulation in ProbTT.

\subsection{Classical AC and Its Problems}

\begin{definition}[Axiom of Choice]
For any family of non-empty sets $\{A_i\}_{i \in I}$, there exists a choice function $f : I \to \bigcup_i A_i$ with $f(i) \in A_i$ for all $i$.
\end{definition}

AC leads to non-constructive results like the Banach-Tarski paradox: a ball can be decomposed and reassembled into two balls of the same size.

\subsection{Graded Choice}

\begin{definition}[Graded AC]
\[
\mathsf{AC}_w : \left(\prod_{i:I} \|A_i\|_{w_i}\right) \to \left\|\prod_{i:I} A_i\right\|_w
\]
where $w$ depends on $I$ and the $w_i$.
\end{definition}

\begin{theorem}[Choice Weight Scaling]
\begin{enumerate}
    \item \textbf{Finite choice}: $w = \one$ (fully constructible)
    \item \textbf{Countable choice}: $w = \prod_{n=1}^\infty w_n$ (may be $< \one$)
    \item \textbf{Uncountable choice}: $w \ll \one$
\end{enumerate}
\end{theorem}

\begin{proof}[Proof sketch]
For finite $I$, we can construct the choice function explicitly---each step preserves weight $\one$, and finitely many steps preserve the product.

For countable $I$, the infinite product $\prod_{n=1}^\infty w_n$ converges but may be strictly less than $\one$ (e.g., if $w_n = 1 - \epsilon/2^n$, then $w < 1$).

For uncountable $I$, the product over uncountably many factors $< \one$ collapses toward $\zero$.
\end{proof}

\subsection{Banach-Tarski at Weight $< \one$}

\begin{theorem}[Graded Banach-Tarski]
The Banach-Tarski decomposition exists at some weight $w_{BT} < \one$.
\end{theorem}

\begin{remark}[Physical Interpretation]
Weight $< \one$ means ``exists mathematically but not physically realizable.'' Banach-Tarski requires infinitely many non-constructive choices; each contributes a factor $< \one$; the product is strictly $< \one$. This correctly predicts that you cannot actually perform the decomposition.
\end{remark}

\subsection{Comparison}

\begin{center}
\begin{tabular}{l|l|l|l}
& \textbf{Constructive} & \textbf{Classical} & \textbf{ProbTT} \\
\hline
AC status & Fails & Holds (weight $\one$) & Holds at weight $w$ \\
BT paradox & Doesn't exist & Exists & Exists at $w < \one$ \\
Information & Binary rejection & Binary acceptance & Graded \\
\end{tabular}
\end{center}

ProbTT provides more information than both: constructive math says ``no choice function''; classical math says ``choice function exists''; ProbTT says ``choice function exists at weight $w$, quantifying constructibility.''

\section{Implementation Status}
%=============================================================================

\subsection{What Is Implemented}

\begin{enumerate}
    \item \textbf{Weight algebra} (\texttt{probtt-impl/src/weight.ml}): Product De Morgan structure with:
    \begin{itemize}
        \item Simplification of weight expressions
        \item Rational arithmetic for exact computation
        \item Fixpoint solver for $w = \negw w \Rightarrow w = \frac{1}{2}$
        \item Unification-based constraint accumulation
    \end{itemize}
    \item \textbf{Bidirectional type checker} (\texttt{probtt-impl/src/check.ml}): Prototype implementing:
    \begin{itemize}
        \item Weight multiplication in elimination rules
        \item Type inference with weight tracking
        \item Weight comparison via $\leq$
    \end{itemize}
    \item \textbf{Proof checker} (\texttt{probtt-impl/src/proof.ml}): Demonstration system for:
    \begin{itemize}
        \item Postulates, derivations, contradictions
        \item Weight forcing from contradictions
        \item Negation fixpoint solving
    \end{itemize}
    \item \textbf{Agda formalization} (\texttt{agda/ProbTT/}): Syntax, judgments, MLTT embedding theorem
\end{enumerate}

\subsection{What Is NOT Implemented}

\begin{enumerate}
    \item \textbf{Weight inference}: The checker does not infer weights from term structure. Weights must be explicitly annotated or default to $\one$. The \texttt{infer\_ctx} in \texttt{weight.ml} accumulates constraints but \texttt{solve\_inference} handles only trivial cases.
    \item \textbf{Constraint propagation}: The Pi-Elim rule should generate $w_{result} = w_f \cdot w_a$, but constraints are not propagated bidirectionally.
    \item \textbf{Dependent weight instantiation}: Weights depending on bound variables (e.g., $w(x)$ in $(x : A) \to B \at{w(x)}$) are parsed but not fully instantiated during application.
    \item \textbf{General proof checking}: The proof checker (\texttt{proof.ml}) demonstrates fixed examples (e.g., $\sqrt{2}$ irrationality). It does not parse arbitrary proof files.
\end{enumerate}

\subsection{Prototype vs Production}

The implementation is a \textbf{prototype} demonstrating the core ideas:
\begin{itemize}
    \item Variables are always assigned weight $\one$ (line 24 of \texttt{check.ml})
    \item Non-trivial fixpoints like $w = w \cdot w$ choose $1$ arbitrarily
    \item The $\sqrt{2}$ proof runs a hardcoded list of declarations, not a parser
\end{itemize}

A production implementation would require bidirectional weight inference, general fixpoint solving, and a proper parser for proof files.
