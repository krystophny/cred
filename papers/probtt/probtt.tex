\begin{abstract}
We define a type theory where judgments carry \emph{weights} from a De Morgan algebra. The algebra has multiplication, complement, zero, one, and order---with disjunction derived via De Morgan duality. When $\W = \{\zero, \one\}$, we recover Martin-L\"of Type Theory. When $\W = {[0,1]}$, we get probabilistic typing with conditioning via the chain rule (not division). The key insight: types and logic \emph{emerge} from weight structure. This is product fuzzy logic lifted to type theory, with conditioning as the key addition.
\end{abstract}

%=============================================================================
\section{The Weight Algebra}
%=============================================================================

\subsection{Product De Morgan Structure}

We axiomatize a minimal structure: multiplication and complement. Disjunction is derived.

\begin{definition}[Weight Algebra]
$\W$ is a type with:
\begin{itemize}
    \item Constants: $\one : \W$ (certainty), $\zero : \W$ (impossibility)
    \item Multiplication: $(\cdot) : \W \to \W \to \W$ (conjunction)
    \item Complement: $\negw : \W \to \W$ (negation)
    \item Order: $(\leq) : \W \to \W \to \Type$
\end{itemize}
\end{definition}

\begin{definition}[Multiplication Axioms]
\begin{align}
w \cdot \one &= w \tag{unit-right} \\
\one \cdot w &= w \tag{unit-left} \\
w \cdot \zero &= \zero \tag{annihilate-right} \\
\zero \cdot w &= \zero \tag{annihilate-left} \\
(u \cdot v) \cdot w &= u \cdot (v \cdot w) \tag{assoc} \\
u \cdot v &= v \cdot u \tag{comm}
\end{align}
\end{definition}

\begin{definition}[Complement Axioms]
\begin{align}
\negw \zero &= \one \tag{neg-zero} \\
\negw \one &= \zero \tag{neg-one} \\
\negw \negw w &= w \tag{involution}
\end{align}
\end{definition}

\begin{definition}[Order Axioms]
\begin{align}
w &\leq w \tag{refl} \\
u \leq v \land v \leq w &\Rightarrow u \leq w \tag{trans} \\
u \leq v \land v \leq u &\Rightarrow u = v \tag{antisym} \\
\zero &\leq w \tag{zero-least} \\
w &\leq \one \tag{one-greatest}
\end{align}
\end{definition}

\subsection{Derived Operations}

\begin{definition}[De Morgan Disjunction]
\[
w \vee v := \negw(\negw w \cdot \negw v)
\]
\end{definition}

When $\W = {[0,1]}$ with $\negw w = 1 - w$, we get:
\[
w \vee v = 1 - (1-w)(1-v) = w + v - wv
\]
This is the \textbf{probabilistic sum} (inclusion-exclusion for independent events).

\begin{theorem}[De Morgan Laws]
\begin{align}
\negw(w \cdot v) &= \negw w \vee \negw v \\
\negw(w \vee v) &= \negw w \cdot \negw v
\end{align}
\end{theorem}

\begin{proof}
Direct calculation from definitions.
\end{proof}

\subsection{What We Have vs What We Don't}

\begin{center}
\begin{tabular}{l|l}
\textbf{Primitive} & \textbf{Derived} \\
\hline
$\cdot$ (multiplication/AND) & $\vee$ (De Morgan OR) \\
$\negw$ (complement/NOT) & $\to$ (implication, if needed) \\
$\zero, \one$ (constants) & \\
$\leq$ (order) &
\end{tabular}
\end{center}

\textbf{NOT primitive}: addition $+$, subtraction $-$, division $/$.

\subsection{Instances}

\begin{example}[Boolean: $\W = \{\zero, \one\}$]
$\one \cdot \one = \one$, else $\zero$. $\negw \zero = \one$, $\negw \one = \zero$.

This is classical Boolean logic.
\end{example}

\begin{example}[Probability: $\W = {[0,1]}$]
$w \cdot v = w \times v$ (ordinary multiplication). $\negw w = 1 - w$.

This is product fuzzy logic / probabilistic reasoning.
\end{example}

\begin{example}[Three-valued: $\W = \{\zero, ?, \one\}$]
$? \cdot ? = ?$, $\negw ? = ?$.

Kleene's strong three-valued logic.
\end{example}

\section{Judgments}
%=============================================================================

\begin{definition}[Judgment Forms]
\begin{enumerate}
    \item $\Gamma \ctx$ \hfill (valid context)
    \item $\Gamma \vdash A \type$ \hfill (valid type)
    \item $\Gamma \vdash a : A \at{w}$ \hfill (weighted typing)
    \item $\Gamma \vdash a \equiv b : A \at{w}$ \hfill (weighted equality)
\end{enumerate}
\end{definition}

\begin{remark}[Interpretation]
$\Gamma \vdash a : A \at{w}$ means: term $a$ inhabits type $A$ with weight $w$.
\begin{itemize}
    \item $w = \one$: $a$ definitely inhabits $A$
    \item $w = \zero$: $a$ does not inhabit $A$ (vacuously true)
    \item $\zero < w < \one$: $a$ partially/probably inhabits $A$
\end{itemize}
\end{remark}

\section{Structural Rules}

\begin{gather}
\infer[\textsc{Ctx-Empty}]
  {\cdot \ctx}
  {}
\qquad
\infer[\textsc{Ctx-Ext}]
  {\Gamma, x : A \ctx}
  {\Gamma \ctx \quad \Gamma \vdash A \type}
\\[1em]
\infer[\textsc{Var}]
  {\Gamma, x : A, \Delta \vdash x : A \at{\one}}
  {}
\\[1em]
\infer[\textsc{Weaken}]
  {\Gamma \vdash a : A \at{v}}
  {\Gamma \vdash a : A \at{w} \quad v \leq w}
\\[1em]
\infer[\textsc{Subst}]
  {\Gamma \vdash b[a/x] : B[a/x] \at{w \cdot v}}
  {\Gamma \vdash a : A \at{w} \quad \Gamma, x : A \vdash b : B \at{v}}
\end{gather}

\section{Type Formers}
%=============================================================================

\subsection{Functions (Dependent)}

\begin{gather}
\infer[\textsc{$\Pi$-Form}]
  {\Gamma \vdash (x : A) \to B \type}
  {\Gamma \vdash A \type \quad \Gamma, x : A \vdash B \type}
\\[1em]
\infer[\textsc{$\Pi$-Intro}]
  {\Gamma \vdash \lambda x. b : (x : A) \to B \at{w}}
  {\Gamma, x : A \vdash b : B \at{w}}
\\[1em]
\infer[\textsc{$\Pi$-Elim}]
  {\Gamma \vdash f\,a : B[a/x] \at{w \cdot v}}
  {\Gamma \vdash f : (x : A) \to B \at{w} \quad \Gamma \vdash a : A \at{v}}
\end{gather}

\textbf{Key rule}: weights multiply in elimination.

\subsection{Pairs (Dependent)}

\begin{gather}
\infer[\textsc{$\Sigma$-Form}]
  {\Gamma \vdash \Sigma(x : A). B \type}
  {\Gamma \vdash A \type \quad \Gamma, x : A \vdash B \type}
\\[1em]
\infer[\textsc{$\Sigma$-Intro}]
  {\Gamma \vdash (a, b) : \Sigma(x : A). B \at{w \cdot v}}
  {\Gamma \vdash a : A \at{w} \quad \Gamma \vdash b : B[a/x] \at{v}}
\\[1em]
\infer[\textsc{$\Sigma$-Elim$_1$}]
  {\Gamma \vdash \pi_1\,t : A \at{w}}
  {\Gamma \vdash t : \Sigma(x : A). B \at{w}}
\qquad
\infer[\textsc{$\Sigma$-Elim$_2$}]
  {\Gamma \vdash \pi_2\,t : B[\pi_1 t / x] \at{w}}
  {\Gamma \vdash t : \Sigma(x : A). B \at{w}}
\end{gather}

Pair weight = product of component weights (joint probability).

\subsection{Sums (Coproducts)}

\begin{gather}
\infer[\textsc{$+$-Form}]
  {\Gamma \vdash A + B \type}
  {\Gamma \vdash A \type \quad \Gamma \vdash B \type}
\\[1em]
\infer[\textsc{$+$-Intro$_L$}]
  {\Gamma \vdash \mathsf{inl}\,a : A + B \at{w}}
  {\Gamma \vdash a : A \at{w}}
\qquad
\infer[\textsc{$+$-Intro$_R$}]
  {\Gamma \vdash \mathsf{inr}\,b : A + B \at{w}}
  {\Gamma \vdash b : B \at{w}}
\end{gather}

\subsection{Empty and Unit}

\begin{gather}
\infer[\textsc{$\mathbf{0}$-Form}]
  {\Gamma \vdash \mathbf{0} \type}
  {\Gamma \ctx}
\qquad
\infer[\textsc{$\mathbf{0}$-Elim}]
  {\Gamma \vdash \mathsf{abort}\,e : A \at{w}}
  {\Gamma \vdash e : \mathbf{0} \at{w}}
\\[1em]
\infer[\textsc{$\mathbf{1}$-Form}]
  {\Gamma \vdash \mathbf{1} \type}
  {\Gamma \ctx}
\qquad
\infer[\textsc{$\mathbf{1}$-Intro}]
  {\Gamma \vdash \star : \mathbf{1} \at{\one}}

\end{gather}

\subsection{Identity}

\begin{gather}
\infer[\textsc{Id-Form}]
  {\Gamma \vdash \mathsf{Id}_A(a, b) \type}
  {\Gamma \vdash a : A \at{\one} \quad \Gamma \vdash b : A \at{\one}}
\\[1em]
\infer[\textsc{Id-Intro}]
  {\Gamma \vdash \mathsf{refl} : \mathsf{Id}_A(a, a) \at{w}}
  {\Gamma \vdash a : A \at{w}}
\end{gather}

\section{Negation}
%=============================================================================

There are two approaches to negation:

\subsection{Type-Theoretic Negation}

\[
\neg A := A \to \mathbf{0}
\]

A proof of $\neg A$ is a function from $A$ to the empty type.

\subsection{Weight Negation}

For a term $a : A \at{w}$, we can consider $\negw w$ as the ``complementary weight.''

In $\W = {[0,1]}$: if $P(A) = w$, then $P(\neg A) = 1 - w = \negw w$.

\begin{remark}[Relationship]
Type-theoretic negation ($A \to \mathbf{0}$) corresponds to weight negation in the following sense: if $a : A \at{w}$ with $w > \zero$, then there is no $f : A \to \mathbf{0} \at{\one}$.
\end{remark}

%=============================================================================
\section{Conditioning (Not Implication)}
%=============================================================================

\subsection{Why Not Material Implication}

Material implication $A \to B := \negw A \vee B$ has paradoxes:
\begin{itemize}
    \item $\mathsf{False} \to \mathsf{Anything} = \mathsf{True}$ (ex falso)
    \item $\mathsf{Anything} \to \mathsf{True} = \mathsf{True}$
\end{itemize}

We use \textbf{conditioning} instead of implication.

\subsection{Chain Rule}

The fundamental equation is:
\[
P(A, B) = P(A \mid B) \cdot P(B) = P(B \mid A) \cdot P(A)
\]

This is \textbf{multiplicative}---no division needed!

\begin{gather}
\infer[\textsc{Chain}]
  {\Gamma \vdash (a, b) : A \times B \at{u \cdot v}}
  {\Gamma \vdash a : A \at{u} \quad \Gamma \vdash b \mid a : B \at{v}}
\end{gather}

The conditional $b \mid a$ is \textbf{primitive}. We don't compute it by division; we assert it exists when the chain rule holds.

\subsection{Conditioning vs Implication}

\begin{center}
\begin{tabular}{l|l|l}
& \textbf{Implication} $A \to B$ & \textbf{Conditioning} $A \mid B$ \\
\hline
Formula & $\negw A \vee B$ & $P(A,B) / P(B)$ \\
Computable from & $A$ and $B$ alone & Joint $P(A,B)$ \\
False antecedent & ``True'' (lies) & Indeterminate (honest) \\
In logic & Yes & No \\
In probability & No & Yes \\
\end{tabular}
\end{center}

\subsection{Graded Ex Falso}

What about $P(A \mid B)$ when $P(B) = 0$?

In classical logic: $\bot \to A$ is always true (ex falso quodlibet).

In ProbTT: when $b : B \at{\zero}$:
\[
(a, b) : A \times B \at{\zero \cdot w} = \at{\zero}
\]
The conditional $a \mid b$ can have \emph{any} weight---the joint is already zero.

\textbf{Key insight}: Ex falso is the $w = 0$ limit of a continuous phenomenon:
\begin{itemize}
    \item $w = \one$: Evidence fully constrains conclusions
    \item $w = \epsilon$: Evidence weakly constrains conclusions
    \item $w = \zero$: Evidence imposes no constraint (ex falso)
\end{itemize}

Classical logic has only the endpoints; ProbTT has the entire spectrum.

\section{MLTT as the $\{\zero, \one\}$ Case}
%=============================================================================

\begin{theorem}[MLTT Embedding]
When $\W = \{\zero, \one\}$, ProbTT reduces to MLTT:
\begin{itemize}
    \item $\Gamma \vdash a : A \at{\one}$ becomes $\Gamma \vdash a : A$
    \item $\Gamma \vdash a : A \at{\zero}$ is vacuously satisfied
    \item Weight operations: $\one \cdot \one = \one$, else $\zero$
    \item Complement: $\negw\zero = \one$, $\negw\one = \zero$
    \item De Morgan OR: $w \vee v = \negw(\negw w \cdot \negw v)$ gives Boolean OR
\end{itemize}
\end{theorem}

\begin{proof}
By inspection. All weight operations compute to Boolean.
\end{proof}

%=============================================================================
\section{Relationship to Fuzzy Logic}
%=============================================================================

\subsection{What We Share}

Our weight algebra is exactly \textbf{Product fuzzy logic}:
\begin{itemize}
    \item Product t-norm: $w \cdot v$ (multiplication)
    \item Standard negation: $\negw w = 1 - w$ (complement)
    \item Probabilistic sum: $w \vee v = w + v - wv$ (De Morgan)
\end{itemize}

This is a well-studied \textbf{De Morgan algebra}.

\subsection{What We Add}

\begin{enumerate}
    \item \textbf{Type theory}: $\Pi$, $\Sigma$, $\mathsf{Id}$, universes
    \item \textbf{Weighted judgments}: $\Gamma \vdash a : A \at{w}$
    \item \textbf{Conditioning}: The chain rule as primitive
    \item \textbf{MLTT as limit}: The $\{\zero, \one\}$ case
\end{enumerate}

Fuzzy logic has no conditioning. We add it via the chain rule.

\subsection{Summary}

\[
\text{ProbTT} = \text{Product Fuzzy Logic} + \text{Type Theory} + \text{Conditioning}
\]

\section{Categorical Semantics}
%=============================================================================

\begin{definition}[Weighted Category]
A \emph{$\W$-weighted category} has:
\begin{itemize}
    \item Objects (types)
    \item For each $w : \W$, morphisms $\mathrm{Hom}_w(A, B)$
    \item Identity: $\mathrm{id}_A \in \mathrm{Hom}_\one(A, A)$
    \item Composition: $f \in \mathrm{Hom}_w(A, B)$, $g \in \mathrm{Hom}_v(B, C)$ gives $g \circ f \in \mathrm{Hom}_{w \cdot v}(A, C)$
\end{itemize}
\end{definition}

When $\W = {[0,1]}$ with copy $\Delta : A \to A \times A$ and delete $! : A \to \mathbf{1}$, this becomes a \textbf{Markov category}.

\section{Summary}
%=============================================================================

\textbf{Weight algebra}: $(\zero, \one, \cdot, \negw, \leq)$ --- Product De Morgan

\textbf{Derived}: $w \vee v = \negw(\negw w \cdot \negw v)$ --- De Morgan OR

\textbf{Judgments}: $\Gamma \vdash a : A \at{w}$

\textbf{Key rule}: Weights multiply in elimination

\textbf{Conditioning}: Chain rule $P(A,B) = P(A|B) \cdot P(B)$, no division

\textbf{MLTT}: The $\W = \{\zero, \one\}$ case

\textbf{Probability}: The $\W = {[0,1]}$ case

\textbf{Novelty}: Product fuzzy logic + type theory + conditioning

\begin{enumerate}

    \item \textbf{Weight inference}: Can weights be inferred from term structure?
    \item \textbf{Decidability}: Type checking decidable if $\W$ decidable
